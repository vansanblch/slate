    Find all unique attribures by query

    Some useful information about the method

    import pl "shared_services/product_list"

req := pl.ProductList.FindUniqAttrsByQueryRequest{
    AttributeUid: []string{"paper_format", "paper_type"},
    Where: []*pl.ProductList.Where{
        &pl.ProductList.Where{
            And: []*pl.ProductList.Where.AttributeSearch{
                &pl.ProductList.Where.AttributeSearch{
                    Uid: "color_type",
                    ValueUid: []string{"4-4", "4-0"},
                },
                &pl.ProductList.Where.AttributeSearch{
                    Uid: "binding_type",
                    ValueUid: []string{"wire-o", "glued"},
                },
            }
        },
        &pl.ProductList.Where{
            And: []*pl.ProductList.Where.AttributeSearch{
                &pl.ProductList.Where.AttributeSearch{
                    Uid: "folding_type",
                    ValueUid: []string{"folded", "tri-fold"},
                },
            }
        },
    },
}

resp, err := NewProductListClient(conn).FindUniqAttrsByQuery(context.Background(), &req)
